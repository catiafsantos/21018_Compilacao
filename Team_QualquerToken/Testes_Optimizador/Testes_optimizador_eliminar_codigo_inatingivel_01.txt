/*eliminar_codigo_inatingivel()          # Remove código após saltos sem label

Os exemplos mais comuns são após return, break, continue, goto (quando não há um label imediatamente a seguir que possa ser alvo de outro goto), ou chamadas a funções que nunca retornam  */


EXEMPLO 1


int obter_valor() {
    int valor = 42;
    printf("A calcular valor...\n");
    return valor; // O controlo sai da função AQUI.

    // ---- INÍCIO DO CÓDIGO INATINGÍVEL ----
    printf("Este printf nunca será executado.\n");
    valor = valor + 10; // Esta atribuição nunca acontece.
    printf("Novo valor (inatingível): %d\n", valor);
    // ---- FIM DO CÓDIGO INATINGÍVEL ----
}

int main() {
    int resultado = obter_valor();
    printf("Valor obtido: %d\n", resultado);
    return 0;
}
